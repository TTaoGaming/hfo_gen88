```typescript
const VirtualToolSchema = z.object({
  id: z.string().uuid(),
  name: z.string(),
  category: z.enum(['surgical', 'construction', 'artistic', 'scientific', 'custom']),
  
  // Form (visual representation)
  form: z.object({
    model3D: z.string().optional(),      // GLB/GLTF URL
    sprite2D: z.string().optional(),     // PNG/SVG URL
    procedural: z.string().optional(),   // Generation prompt
    scale: z.number().default(1),
    anchor: Point3DSchema,               // Where hand grips
  }),
  
  // Function (interaction model)
  function: z.object({
    primaryAction: z.string(),           // "cut", "grip", "paint"
    gestures: z.record(z.string()),      // gesture → action mapping
    constraints: z.object({
      minForce: z.number().optional(),
      maxAngle: z.number().optional(),
      requiredPose: z.string().optional(),
    }),
  }),
  
  // Feedback (training integration)
  feedback: z.object({
    successCriteria: z.array(z.string()),
    dangerZones: z.array(BoundingBoxSchema),
    guidancePrompts: z.array(z.string()),
  }),
});
```

```typescript
async function materializeTool(intent: string): Promise<VirtualTool> {
```
